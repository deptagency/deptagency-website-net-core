{# An example button macro. Using macro's for buttons is very useful as it keeps the implementation of a button consistent and easily maintainable as you only have a single place where it's defined. Usage: {% include 'elements/button.html' %} {{ button( { label: 'Click here' } ) }} {{ button( { label: 'Tam Tam', url: 'http://www.tamtam.nl', class: 'js-tamtam-button' } ) }} #} {% macro button( data ) %} {{ data.label or 'label' }} {% endmacro %}
